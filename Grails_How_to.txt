// Create a Grails project
grails create-app <project_name>

// Create a Domain class
// Domain class stores persistance data.
cd <project_name>
grails create-domain-class org.example.ClassName

	package org.example
	class ClassName {
	// Define persitstance data
	    String data1
	    String data2
	// Define validation requirements    	
	    static constraints = {
	        data1(blank: false)
	        data1(blank: false)
	    }
	}

// Create a Controller 
grails create-controller org.example.ClassName // Note the capital C

	package org.example
	class BookController {
		// Enable scaffolding
	    def scaffold = ClassName
	}

// Start Grails
grails run-app
